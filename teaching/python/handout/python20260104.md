```python
import os
from os.path import basename, exists

def download(url):
    filename = basename(url)
    if not exists(filename):
        from urllib.request import urlretrieve

        local, _ = urlretrieve(url, filename)
        print("Downloaded " + str(local))
    return filename

def extract_text(input_file, output_file, start_kw, end_kw):
    with open(input_file, 'r', encoding='utf-8') as f_in, \
         open(output_file, 'w', encoding='utf-8') as f_out:
        
        recording = False
        for line in f_in:
            # Check if we hit the starting keyword
            if start_kw in line:
                recording = True
                continue  # Skip the line containing the keyword itself
            
            # Check if we hit the ending keyword
            if end_kw in line:
                recording = False
                break  # Stop reading once the end keyword is found
            
            # Write to output if we are in the recording state
            if recording:
                f_out.write(line)
    
    f_in.close()
    f_out.close()

if __name__ == '__main__':
    download('https://www.gutenberg.org/cache/epub/345/pg345.txt')
    extract_text('pg345.txt', 'pg345_cleaned.txt', '*** START', '*** END')
```

- The items in a Python dictionary are stored in a **hash table**, which is a way of organizing data that has a remarkable property: **the in operator takes about the same amount of time no matter how many items are in the dictionary**.
```python
import time

def reverse_word(word):
    return ''.join(reversed(word))

def too_slow():
    count = 0
    for word in word_list:
        if reverse_word(word) in word_list:
            count += 1
    return count

if __name__ == '__main__':
    word_list = open('words.txt').read().split()

    start_time = time.perf_counter()
    total = 0
    for i in range(100000):
        total += i * i

    result = too_slow()
    print('The total of count reversed words: ', result)

    end_time = time.perf_counter()

    elapsed_time = end_time - start_time
    print(f"\nExecution time: {elapsed_time:.4f} seconds")
```

```python
#filename: much_faster.py
```

## Accumulating a list
- we’ll loop through the words in **word_dict** and make a list of palindromes – that is, words that **are spelled the same backward and forward, like “noon” and “rotator”**.
```python
def reverse_word(word):
    return ''.join(reversed(word))

def is_palindrome(word):
    """Check if a word is a palindrome."""
    return reverse_word(word) == word

if __name__ == '__main__':
    word_list = open('words.txt').read().split()
    word_dict = {}
    for word in word_list:
        word_dict[word] = 1

    count = 0
    palindromes = []
    long_palindromes = []
    for word in word_dict:
        if  is_palindrome(word):
            count +=1
            palindromes.append(word)
        
    for word in palindromes:
        if len(word) >= 7:
            long_palindromes.append(word)
            
    print('palindrome count: ', count)
    print('\npalindromes[:10]: ', palindromes[:10])
    print('\npalindrome >=7 is: ', long_palindromes)
```
